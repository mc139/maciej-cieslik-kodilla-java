SELECT ISSUESLIST_ID,SUMMARY,USER_ID_ASSIGNEDTO, NAME
FROM ISSUES I
JOIN ISSUESLISTS L
ON I.ISSUESLIST_ID = L.ID;


SELECT FIRSTNAME,LASTNAME,SUMMARY,DESCRIPTION
FROM ISSUES I
JOIN USERS U
ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT FIRSTNAME,LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY USER_ID_ASSIGNEDTO
HAVING COUNT(*)>0;
